#include <bits/stdc++.h>
using namespace std;

struct Node {
	int data;
	struct Node * next;
	// Function for creating a node
	Node(int data) {
		this->data = data;
		next = NULL;
	}
};

struct Graph {

	vector<Node*> adjList;

	 	numVertices = vertices;

};

// Function for adding edges
void addEdges(Graph * graph, int u, int v) {
	// adding u -> v
	Node * newnode = new Node(v);
	newnode->next = graph->adjList[u];
	graph->adjList[u] = newnode;
	
	// adding v -> u;
	Node * newnode1 = new Node(u);
	newnode1->next = graph->adjList[v];
	graph->adjList[v] = newnode1;
}

// Function to print the graph normally in order to verify whether the connections are kept properly or not
void printList(Graph * graph) {
	for (int i = 1; i <= graph->numVertices; i++) {
		Node * temp = graph->adjList[i];
        cout << "The Adjacency List of vertex " << i << " -> ";
		while (temp) {
			if (temp->next)
			cout << temp->data << " -> ";
			else cout << temp->data;
			temp = temp->next;
		}
		cout << endl;
	}
}

// DFS of graph
void dfs(Graph * graph, int start, vector <int> freq1) {
	freq1[start] = 1;
	cout << start << " ";
	Node * temp = graph->adjList[start];
	while (temp) {
		if (!freq1[temp->data]) dfs(graph, temp->data, freq1);
		temp = temp->next;
	}
}

// BFS of graph
void bfs(vector<int>graph[], int start, vector<int> freq2) {
    queue<int> q;
    q.push(start);
    freq2[start] = 1;
    while (!q.empty()) {
        int cur = q.front();
        q.pop();
        cout << cur << " ";
        for(int i : graph[cur]){
        	if(freq2[i] != 1){
        		q.push(i);
        		freq2[i] = 1;
			}
		}
    }
}
